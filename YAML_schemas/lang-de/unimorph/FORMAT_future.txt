      applies: {u-Aspect: [u-prf]}

  Rules:
    de-Tense-type_de-futur-I:
      applies: {de-Tense-type: de-futur-I}

    de-Tense-type_de-futur-II:
      applies: {de-Tense-type: de-futur-II}
      pattern: "werden (finite) + past participle + haben/sein infinitive"
      examples: 
        - "ich werde gegangen sein"

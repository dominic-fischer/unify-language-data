ReportedSpeech:
  Features:
    u-Evidentiality: [u-fh, u-quot]
    u-Interrogativity: [u-decl, u-int, x-excl]
    u-Tense: [u-prs, u-pst, u-fut]

  Notes:
  - "The tense of the subordinate clause remains unchanged unless the reporting verb is in the past, in which case the reported tense is adjusted relative to it as outlined."
  - "In either case, though, mood changes: Subjunctive I is the mood of reporting, Subjunctive II if forms coincide"
  - "Using 'dass' will trigger the inversion, as the sentence then becomes a subclause."

  Negation:
    applies: {x-Negation-type: x-clause}
    note: "Regular negation."
    examples:
    - "Er sagte, er wolle den Kuchen nicht essen."

  Rules:
    u-Evidentiality_u-fh:
      applies: {u-Evidentiality: u-fh}
      pattern: "quote + _reporting verb_"
      example: "Er sagte: 'Ich komme morgen.'"
      note: "quotation preserves original tense/mood exactly"

    u-Evidentiality_u-quot__u-Interrogativity_u-int:
      applies: {u-Evidentiality: u-quot, u-Interrogativity: u-int}
      pattern: "_reporting verb_ + ob/wh-word + {clause in Subjunctive I (present/perfect/future)}"
      example: "Er fragte, ob er morgen komme."
      note: "yes/no questions use 'ob', wh-questions use wh-word; tense follows same system as u-decls (pres, past, fut)"

    u-Evidentiality_u-quot__u-Interrogativity_x-excl:
      applies: {u-Evidentiality: u-quot, u-Interrogativity: x-excl}
      pattern: "_reporting verb_ + wie/was für + {clause in Subjunctive I (present/perfect/future)}"
      example: "Er sagte, wie schön es sei!"
      note: "exclamatives follow the same tense distinctions as u-decls."
    u-Evidentiality_u-quot__u-Interrogativity_u-decl__u-Tense_u-prs:
      applies: {u-Evidentiality: u-quot, u-Interrogativity: u-decl, u-Tense: u-prs}
      pattern: "_reporting verb_ + dass/Ø + {clause in Subjunctive I (present)}"
      examples:
      - "Er sagte, er komme morgen."
      - "Er sagte, dass er morgen komme."

    u-Evidentiality_u-quot__u-Interrogativity_u-decl__u-Tense_u-pst:
      applies: {u-Evidentiality: u-quot, u-Interrogativity: u-decl, u-Tense: u-pst}
      pattern: "_reporting verb_ + dass/Ø + {clause in Subjunctive I (perfect)}"
      example: "Er sagte, er sei gestern angekommen."

    u-Evidentiality_u-quot__u-Interrogativity_u-decl__u-Tense_u-fut:
      applies: {u-Evidentiality: u-quot, u-Interrogativity: u-decl, u-Tense: u-fut}
      pattern: "_reporting verb_ + dass/Ø + {clause in Subjunctive I (future)}"
      example: "Er sagte, er werde morgen kommen."
      note: "reported future"

Features:
  u-Person: [u-1, u-2, u-3]
  u-Number: [u-sg, u-pl]
  u-Tense: [u-prs]

Rules:
  U-Tense-u-prs:
    applies: { u-Tense: u-prs }
    pattern: "Ndi-ka-<stem>"
    note: "Present conditional in Shona; -ka- marks the conditional; -chi- can substitute -ka- in the positive; -sa- is inserted after -ka- for negation."
    examples:
    - "Ndi-ka-siya ndi-no/cha/nga-rasika."
    - "Ndi-ka-sa-siya ha-ndi-goni kurasika."

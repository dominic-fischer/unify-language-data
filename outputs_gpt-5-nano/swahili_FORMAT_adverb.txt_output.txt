Adverbs:
  Features:
    u-Part-of-Speech: [u-adv]
  Rules:
    u-Part-of-Speech_u-adv:
      applies: {u-Part-of-Speech: u-adv}
      patterns:
        - pattern: "kwa <Nomen>"
          examples:
            - "Anakimbia kwa kasi."
          note: "Nominal adverb formed from noun with the preposition kwa."
        - pattern: "bila <Nomen>"
          examples:
            - "Anakimbia bila kasi."
          note: "Negation of nominal adverb using bila."
        - pattern: "vi- + adj stem"
          examples:
            - "Tunafanya hiki vizuri."
          note: "Adjectival adverbs formed with the vi- prefix."
        - pattern: "haraka (adjective used as adverb without modification)"
          examples:
            - "Mbweha hukimbia haraka."
          note: "Some adjectives can be used adverbially without morphological change."

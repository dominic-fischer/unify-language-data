    u-Gender_u-masc:
      applies: {u-Gender: u-masc}
      patterns:
        - pattern: "dictionary form often ends in a consonant or -u"
          examples:
            - "om"
            - "bou"
            - "copac"
        - pattern: "plural often ends in -i"
          examples:
            - "pom → pomi"
            - "cal → cai"
            - "tată → tați"
      note: "Nouns denoting male people are masculine regardless of ending (e.g., tată, popă)."

    u-Gender_u-fem:
      applies: {u-Gender: u-fem}
      patterns:
        - pattern: "dictionary form often ends in -ă or -a"
          examples:

        - pattern: "plural often ends in -e"
          examples:
            - "masă → mese"
            - "stea → stele"
            - "cafea → cafele"
      note: "Nouns denoting female people are feminine regardless of ending."

    u-Gender_u-neut:
      applies: {u-Gender: u-neut}
      patterns:
        - pattern: "behaves like masculine in the singular and like feminine in the plural"
          examples:
            - "un corp, două corpuri"
            - "un sertar, două sertare"
        - pattern: "dictionary form often ends in a consonant or -u"
          examples:
            - "drum"
            - "cadou"
            - "exemplu"
        - pattern: "plural often ends in -e (also -uri occurs)"
          examples:
            - "teatru → teatre"
            - "muzeu → muzee"
            - "tren → trenuri"
      note: "In synchronic terms, neuter can be analysed as 'ambigeneric' (masculine singular, feminine plural)."

Subjunctive:
  Features:
    u-Mood: [u-sbjv]
    u-Polarity: [u-pos, u-neg]
    ny-Subjunctive-infix: [ny-go-ka, ny-come-dza, ny-neg-habit-ma, ny-modal-nga]
    x-Process: [x-prefixation, x-suffixation]

  Usage:
    polite-requests-and-wishes:
      applies: {u-Mood: u-sbjv}
      note: "Used for polite requests and wishes."
      examples:
      - "Mundipatse mkhate."

    after-kuti-after-verbs-of-wishing-wanting-requesting:
      applies: {u-Mood: u-sbjv}
      note: "Used after (kuti) 'that' following verbs of wishing, wanting, and requesting/commanding."
      examples:
      - "Tikufuna (kuti) mubwere kwathu."

    after-kuti_purpose-result:
      applies: {u-Mood: u-sbjv}
      note: "Used after kuti in the sense of 'so that/so that... (result/purpose)'."
      examples:
      - "Tipita lero kuti tifike posachedwa."
      - "Tipita lero kuti tisachedwe."

    modal-nga_for_could-would:
      applies: {u-Mood: u-sbjv, ny-Subjunctive-infix: ny-modal-nga}
      note: "If meanings like 'could' or 'would' are intended, the modal infix -nga- is additionally present."
      examples:
      - "Tingapite mawa?"

  Rules:
    u-Mood_u-sbjv:
      applies: {u-Mood: u-sbjv}
      pattern: "_SUBJ(SubjectPrefix)_ + _VerbStem_ + e"
      note: "Unlike the indicative, verbs in the subjunctive end in -e."
      examples:
      - "Ndithandizé."

    u-Mood_u-sbjv__u-Polarity_u-neg:
      applies: {u-Mood: u-sbjv, u-Polarity: u-neg}
      pattern: "_SUBJ(SubjectPrefix)_ + sa + _VerbStem_ + e"
      note: "For negation, -sa- can be inserted after the subject prefix."
      examples:
      - "Ndisathandíze."

    u-Mood_u-sbjv__ny-Subjunctive-infix_ny-go-ka:
      applies: {u-Mood: u-sbjv, ny-Subjunctive-infix: ny-go-ka}
      pattern: "_SUBJ(SubjectPrefix)_ + ká + _VerbStem_ + e"
      examples:
      - "Ndikáthandizé."

    u-Mood_u-sbjv__ny-Subjunctive-infix_ny-come-dza:
      applies: {u-Mood: u-sbjv, ny-Subjunctive-infix: ny-come-dza}
      pattern: "_SUBJ(SubjectPrefix)_ + dzá + _VerbStem_ + e"
      examples:
      - "Ndidzáthandizé."

    u-Mood_u-sbjv__ny-Subjunctive-infix_ny-neg-habit-ma:
      applies: {u-Mood: u-sbjv, ny-Subjunctive-infix: ny-neg-habit-ma}
      pattern: "_SUBJ(SubjectPrefix)_ + má + _VerbStem_ + e"
      examples:
      - "Ndimáthandizé."

    u-Mood_u-sbjv__ny-Subjunctive-infix_ny-modal-nga:
      applies: {u-Mood: u-sbjv, ny-Subjunctive-infix: ny-modal-nga}
      pattern: "_SUBJ(SubjectPrefix)_ + nga + _VerbStem_ + e"
      examples:
      - "Tingapite mawa?"

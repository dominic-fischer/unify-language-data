Adverbs:
  Features:
    x-Process: [x-suffixation]

  Rules:
    x-Process_x-suffixation:
      applies: {x-Process: x-suffixation}
      patterns:
      - pattern: "_adjective.F.SG_ + mente"
        examples:
        - "lenta → lentamente"
        note: "Derived by adding -mente to the feminine singular form of the adjective."
      - pattern: "_adjective in -re/-le (F.SG, final -e dropped)_ + mente"
        examples:
        - "facile → facilmente"
        - "particolare → particolarmente"
        note: "Adjectives ending in -re or -le lose final -e before adding -mente."
      - pattern: "_absolute superlative adjective_ + mente"
        examples:
        - "lentissimamente"
        - "facilissimamente"
        note: "These adverbs can also be derived from the absolute superlative form of adjectives."

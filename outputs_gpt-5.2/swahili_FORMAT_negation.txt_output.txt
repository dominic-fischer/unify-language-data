Negation:
  Features:
    x-Negation-type: [x-clause, x-constituent]

  Rules:
    x-Negation-type_x-clause:
      applies: {x-Negation-type: x-clause}
      patterns:
        - pattern: "ha- + subject marker + verb stem"
          examples:
            - "Ha-ja-penda."
          note: "General negative prefix is ha-. For class 1/2 singular persons there are changes (see below)."
        - forms:
            - features: {u-Person: u-1, u-Number: u-sg}
              form: "si-"
              note: "Class 1/2: replaces ha- + ni."
            - features: {u-Person: u-2, u-Number: u-sg}
              form: "hu-"
              note: "Class 1/2: replaces ha- + u."
            - features: {u-Person: u-3, u-Number: u-sg}
              form: "ha-"
              note: "Class 1/2: replaces ha- + a."
          note: "These person-specific changes are marked [V] in the source."
        - pattern: "kennsilbe + si + object marker + verb stem ending in -e"
          examples:
            - "Usinipe mkate!"
          note: "Negative polite requests are formed with the negative subjunctive."
        - pattern: "... si ... (subjunctive)"
          examples:
            - "Tunaondoka leo (ili) tusichelewe kufika."
          note: "Subjunctive negation is formed by inserting -si-."

    x-Negation-type_x-constituent:
      applies: {x-Negation-type: x-constituent}
      patterns:
        - pattern: "bila + noun"
          examples:
            - "Yeye ni mtu bila heshima."
          note: "Nominal adjective negation."
        - pattern: "bila + noun"
          examples:
            - "Anakimbia bila kasi."
          note: "Nominal adverb negation."
        - pattern: "replace -li- with -si-"
          examples:
            - "Dirisha lisilofungwa limevunjika."
          note: "Used for negation of verbal adjectives."

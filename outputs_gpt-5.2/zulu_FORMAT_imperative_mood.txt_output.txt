  Features:
    u-Person: [u-2]
    u-Number: [u-sg, u-pl]
    u-Politeness: [u-infm, u-pol]
    u-Polarity:
    - u-pos
    x-Pronoun-type:
    - x-pers

  Rules:
    u-Number_u-sg__u-Person_u-2__u-Polarity_u-pos:
      applies: {u-Number: u-sg, u-Person: u-2, u-Polarity: u-pos}
      patterns:
      - pattern: "verb stem"
        notes:
        - "used when addressing one lower-ranking person"
      - pattern: "(yi- / y- / wo-) + verb stem"
        notes:
        - "yi- is used when the verb stem has one syllable"
        - "y- is used when the verb stem begins with a vowel"
        - "exception: verb -za 'to come' uses wo- (woza)"
        examples:
        - "Yidlá!"
        - "Yenzá!"
        - "Síza!"

    u-Number_u-pl__u-Person_u-2__u-Polarity_u-pos:

      patterns:
      - pattern: "verb stem + -ni"
        notes:
        - "used when addressing multiple people"
        - "also used when addressing a higher-ranking person (respect form)"
      - pattern: "(yi- / y- / wo-) + verb stem + -ni"
        notes:
        - "yi- is used when the verb stem has one syllable"
        - "y- is used when the verb stem begins with a vowel"
        - "exception: verb -za 'to come' uses wo- (wozani)"
        examples:
        - "Yidláni!"
        - "Yenzáni!"
        - "Sizáni!"

Subjunctive:
  Features:
    u-Mood: [u-sbjv]
    u-Tense: [u-prs, u-pst]
    u-Polarity: [u-pos, u-neg]

  Usage:
    Polite-requests-wishes:
      applies: {u-Mood: u-sbjv}
      note: "Used for polite requests and wishes."
      examples:
      - "Ngicela ningiphe isinkwa."

    After-verbs-of-wanting-wishing-requesting:
      applies: {u-Mood: u-sbjv}
      note: "Used after verbs of wishing, wanting, and requesting/ordering."
      examples:
      - "Wayefuna ngimnike isinkwa"

    After-ukuze-ukuthi-purpose-result:
      applies: {u-Mood: u-sbjv}
      note: "Used after ‘ukuze’/‘ukuthi’ in the sense of ‘(so) that / in order that’."
      examples:
      - "Siyahamba namuhla ukuze sifike maduze."

    Hypothetical-wishes-requests-consecutive-complements:
      applies: {u-Mood: u-sbjv}
      note: "Expresses hypothetical situations, wishes and requests; also used for consecutive action, and as a complement to certain auxiliary verbs and conjunctions."
      examples:
      - "Wāvuka wagqoka wahamba."
      - "Woza lapha uzame futhi!"

  Rules:
    u-Mood_u-sbjv__u-Polarity_u-pos__u-Tense_u-prs:
      applies: {u-Mood: u-sbjv, u-Tense: u-prs, u-Polarity: u-pos}
      pattern: "[subj2][obj]...e"
      note: "Unlike the indicative, subjunctive verbs end in -e."
      examples:
      - "Ngicela ningiphe isinkwa."

    u-Mood_u-sbjv__u-Polarity_u-neg__u-Tense_u-prs:
      applies: {u-Mood: u-sbjv, u-Tense: u-prs, u-Polarity: u-neg}
      pattern: "[subj2]nga[obj]...i"
      examples:
      - "Ubaba ufuna ukuthi ngingahambi lapho."

    u-Mood_u-sbjv__u-Polarity_u-pos__u-Tense_u-pst:
      applies: {u-Mood: u-sbjv, u-Tense: u-pst, u-Polarity: u-pos}
      pattern: "[subj1]a[obj]...a"
      note: "Affirmative past subjunctive is identical to the affirmative remote past, except that the infix -a- is short; in regular Zulu spelling this distinction is not marked."

    u-Mood_u-sbjv__u-Polarity_u-neg__u-Tense_u-pst:
      applies: {u-Mood: u-sbjv, u-Tense: u-pst, u-Polarity: u-neg}
      patterns:
      - pattern: "a[subj1]anga[obj]...a"
        note: "Uncontracted form; alternative negative past forms are contracted to different degrees."
      - pattern: "[subj1]anga[obj]...a"
        note: "Contracted variant."
      - pattern: "a[subj1]a[obj]...a"
        note: "Contracted variant."

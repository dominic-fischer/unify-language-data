    x-Position_x-postnominal:
      applies:
        x-Position: x-postnominal
      pattern: NOUN + (CONCORD matching noun class) + ADJECTIVE
      examples:
      - Ikati elihle.
      - Inja enhle.
      note: Adjectives follow the noun they modify and require a prefix (concord) whose class matches the noun.
    x-Concord-type_x-copulative-concord:

    x-Adjective-type_x-nominal:
      applies:
        x-Adjective-type: x-nominal
      patterns:
      - pattern: na/ne/no + NOUN
        examples:
        - Ungumuntu unomusa.
      - pattern: nge + na/ne/no + NOUN
        examples:
        - Ungumuntu ungenomusa.
      note: Nominal adjectives are formed with a marker (Kennsilbe) plus a noun.
    x-Adjective-type_x-verbal:
      applies:
        x-Adjective-type: x-verbal
      patterns:
      - pattern: adjective syllable + verb stem
        examples:
        - lo ngumsebenza olula
        note: Verbs can be used adjectivally using the 'property syllable' (Eigenschaftssilbe).
      - pattern: subject + marker (Kennsilbe) + verb stem (predicative)
        examples:
        - Umsebenzi ulula.
        note: If the predicative adjective is a verb describing a state, the marker (Kennsilbe) is used.
    u-Polarity_u-neg:
      applies:
        u-Polarity: u-neg
      pattern: a- + Kennsilbe(V) + predicative adjective
      note: 'Exception: in noun class 7/8 the Kennsilbe(V) changes to -ka-.'

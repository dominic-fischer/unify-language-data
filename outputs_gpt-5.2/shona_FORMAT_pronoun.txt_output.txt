  Usage:
    u-Politeness_u-pol:
      applies:
        u-Politeness: u-pol
      note: Adults should always be addressed in 2nd person plural; singular is considered impolite in normal usage and is only used towards children or clearly lower-ranked persons.
  Rules:
    x-Pronoun-type_x-pers:
      applies:
        x-Pronoun-type: x-pers
      note: In Shona, personal pronouns are identical for subject and object.
      forms:
      - features:
          u-Person: u-1
          u-Number: u-sg
        form: ini
      - features:
          u-Person: u-2
          u-Number: u-sg
        form: iwe
      - features:
          u-Person: u-3
          u-Number: u-sg
        form: iye
      - features:
          u-Person: u-1
          u-Number: u-pl
        form: isu
      - features:
          u-Person: u-2
          u-Number: u-pl
        form: imwi
      - features:
          u-Person: u-3
          u-Number: u-pl
        form: ivo
    x-Pronoun-type_x-dem:
      applies:
        x-Pronoun-type: x-dem
      note: Shona distinguishes four types of demonstratives; demonstrative forms vary by noun class.
      patterns: []
    x-Pronoun-type_x-int:
      applies:
        x-Pronoun-type: x-int
      note: To ask which one is meant, use the noun-class marker and add -pi.
      patterns:
      - pattern: _class-marker_ + pi
        examples:
        - api?
        - vapi?
        - upi?
        - ipi?
        - ripi?
        - zvipi?
        - dzipi?
        - rwupi?
        - hwupi?
        - kapi?
        - twupi?
        - kupi?
        - zipi?
        - papi?
        - mupi?
    x-Pronoun-type_x-rel:
      applies:
        x-Pronoun-type: x-rel
      pattern: relative constructions are formed with the possessive syllable attached before the verb
      examples:
      - Murume wa-ndaka-(mu-)ona aka-tiza.
      - Zvandichataura.

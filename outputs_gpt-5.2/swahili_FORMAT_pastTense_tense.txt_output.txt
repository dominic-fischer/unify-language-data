Past:
  Features:
    sw-TAM:
    - sw-past
    - sw-past-neg
    - sw-perfect
    - sw-perfect-neg
    - sw-pluperfect
    - sw-pluperfect-neg
    - sw-imperfect
    - sw-imperfect-neg
    - sw-perfect-continuous
    - sw-perfect-continuous-neg
    - sw-situational
    - sw-consecutive-past

  Usage:
    sw-past:
      note: "Used to talk about actions or states in the past, whether near or distant."
      examples:
      - "Alikwenda."
      - "Nilikula."
    sw-perfect:
      note: "Indicates a past action/situation with relevance to the present moment."
      examples:
      - "Amekwenda."
      - "Nimekula."
    sw-perfect-neg:
      note: "Some sources describe the negative perfect with -ja- as implying 'not yet'; this can be made explicit with bado."
      examples:
      - "Hajaenda."
      - "Sijala."
    sw-situational:
      note: "Used for a simultaneous/background action or situation in subordinate clauses (roughly 'if/when')."
      examples:
      - "Ni-ki-enda."
    sw-consecutive-past:
      note: "In narratives with multiple past events, the first verb commonly takes -li- and following verbs take -ka- (roughly 'and then')."
      examples:
      - "Ni-li-m-piga simu, ni-ka-tembea kwake, ni-ka-fanya kila kitu."
      - "Tulicheza soka tukashinda."
      - "Tulicheza soka na kushinda."

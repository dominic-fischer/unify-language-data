Agreement:
  Features:
    x-Domain:
    - x-subject-verb
    - x-noun-adjective
    - x-noun-determiner

  Rules:
    x-Domain_x-subject-verb:
      applies: {x-Domain: x-subject-verb}
      pattern: "verb takes a subject prefix (subject marker) agreeing with the subject (noun class or person)"
      examples:
      - "A-nthu a-ku-pita ku-nkhalango."
      - "Koma ine ndi-ku-funa kupita uko!"
      - "mwaná ápita; aná ápita"
      note: "personal pronouns (ine, iwe, iye, ife, inu, iwo) are used only for emphasis; otherwise person subject markers are used"

    x-Domain_x-noun-determiner:
      applies: {x-Domain: x-noun-determiner}
      patterns:
      - pattern: "demonstratives agree with the noun class of the head noun"
        examples:
        - "Uyu ndi mwaná wángá."
        - "Awa ndi aná ángá."
        - "Ichi ndi chímanga chánga."
        - "Iyi ndi nyumbá yángá."
        - "mwaná uyu (mwanáyu); aná awa (anáwa)"
        - "mwaná uyo (mwanáyo); aná awo (anáwo)"
        note: "shortened forms in parentheses are more common"
      - pattern: "in classes 1/2, demonstratives can be used in 'who is this/these' questions"
        examples:
        - "uyu ndaní?"
        - "awa ndaní?"
      - pattern: "class 2 (plural of class 1) may be used as respectful concord with singular elders"
        examples:
        - "bambo anga akuyenda, ndikuwaona"
        note: "using the singular object marker -mu- in such contexts is noted as grossly impolite"

    x-Domain_x-noun-adjective:
      applies: {x-Domain: x-noun-adjective}
      patterns:
      - pattern: "possessive adjectives agree with noun class (using concords such as w-/a- in classes 1/2)"
        examples:
        - "Uyu ndi mwaná wángá."
        - "Awa ndi aná ángá."
      - pattern: "genitive/associative marker 'á/of' agrees with noun class (e.g. wá ~ á in classes 1/2)"
        examples:
        - "mwaná wá Mphátso"
        - "aná á Mphátso"
        - "Mwana wa mkazi a-ku-dya."
      - pattern: "verbal adjectives can be formed with wá + ku- (often shortening to wó-)"
        examples:
        - "mwaná wókóngola; aná ókóngola"
        - "mwaná wákúbá; aná ákúbá"
      - pattern: "some adjectives take a double prefix combining possessive concord and numeral concord"
        examples:
        - "mwaná wáḿkúlu; aná áákúlu"
        - "mwaná wáḿng'óno; aná ááng'óno"
        - "mwaná wámwámúna; aná áámúna"
        - "mwaná wáḿkázi; aná áákázi"

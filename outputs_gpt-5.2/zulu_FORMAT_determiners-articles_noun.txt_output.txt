    u-Case_u-loc:
      notes:
      - "Locatives can have translations such as 'in', 'at', 'on', 'to' or 'from'."
      applies: {u-Case: u-loc}
      patterns:
      - title: "Classes 1(a) and 2(a)"
        pattern: "kú- + noun (drop augment)"
        notes:
        - "For nouns in class 1(a) or 2(a), the locative is formed by prefixing kú- to the noun and dropping the augment."
        examples:
        - "umúntu → kumúntu"
        - "abántu → kubántu"
        - "úbabá → kúbabá"
        - "ṓbabá → kṓbabá"
        - "u-Okthoba → ku-Okthoba"
      - title: "Other classes (augment replacement + suffix)"
        pattern: "replace augment with e- (or o- in class 11) (+ usually add -ini with vowel alternations)"
        notes:
        - "For nouns not in class 1 or 2, replace the augment with e-, or with o- for class 11 nouns."
        - "Most nouns also take a locative suffix -ini, triggering the following changes: a+ini→eni; e+ini→eni; i+ini→ini; o+ini→weni; u+ini→wini."
        examples:
        - "uḿlenze → emlénzeni"
        - "îsó → êswéni"
        - "ámánzi → émánzini"
        - "índlebé → éndlebéni"
        - "ínkungú → énkungwíni"
        - "ū́phahla → ōpháhleni"
      - title: "Locative without suffix"
        pattern: "augment replacement only (no -ini)"
        notes:
        - "Some nouns form locatives without the suffix, using just the prefix; this includes most place names and some regular nouns."
        examples:
        - "îGóli → êGóli"
        - "îndlé → êndlé"
        - "îkhânda → êkhânda"
        - "îkhâya → êkhâya"

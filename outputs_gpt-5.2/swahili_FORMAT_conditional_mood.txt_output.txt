Mood:
  Features:
    u-Mood: [u-imp, u-sbjv]
    u-Polarity: [u-pos, u-neg]
  Rules:
    u-Mood_u-imp__u-Polarity_u-pos:
      applies: {u-Mood: u-imp, u-Polarity: u-pos}
      patterns:
      - pattern: "STEM-a"
        note: "Imperative without object prefix."
        examples:
        - "Kunywa!"
        - "Angalia!"
        - "Saidia!"
        - "Jibu!"
      - pattern: "[OBJ]-STEM-e"
        note: "With an object prefix, final -a becomes -e."
        examples:
        - "Yanywe!"
        - "Mwangalie!"
        - "Nisaidie!"
        - "Nijibu!"
        - "Jifunze!"
      - pattern: "STEM-eni / STEM-ni"
        note: "Plural imperative uses the suffix -ni (realised as -eni/-ni depending on verb type)."
        examples:
        - "Kunyweni!"
        - "Angalieni!"
        - "Saidieni!"
        - "Jibuni!"
        - "Jifunzi-ni!"
    u-Mood_u-sbjv__u-Polarity_u-pos:
      applies: {u-Mood: u-sbjv, u-Polarity: u-pos}
      note: "Subjunctive is marked by absence of a TAM prefix and final -e (where applicable)."
      pattern: "[SUBJ]-(OBJ)-STEM-e"
      examples:
      - "Aende."
      - "Nile."
      - "Uandike."
      - "Tusahau."
    u-Mood_u-sbjv__u-Polarity_u-neg:
      applies: {u-Mood: u-sbjv, u-Polarity: u-neg}
      note: "Negative subjunctive inserts -si- in the TAM slot; uses the positive subject prefix."
      pattern: "[SUBJ]-si-(OBJ)-STEM-e"
      examples:
      - "Asiende."
      - "Nisile."
      - "Usiandike."
      - "Tusisahau."
      - "Usitoke."

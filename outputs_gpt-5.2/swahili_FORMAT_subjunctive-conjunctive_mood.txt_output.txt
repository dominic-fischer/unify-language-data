Subjunctive:
  Features:
    u-Mood: [u-sbjv]
    u-Polarity: [u-pos, u-neg]

  Usage:
    PoliteRequestsAndWishes:
      applies: {u-Mood: u-sbjv}
      note: "Used for polite requests and wishes; also as a more polite alternative to an imperative with 2nd person subject."
      examples:
      - "Tafadhali (u)nipe mkate."
      - "Utoke."
    ComplementAfterWishingWantingRequesting:
      applies: {u-Mood: u-sbjv}
      note: "Used after verbs expressing wishes, wanting, suggestions, recommendations, and requests."
      examples:
      - "Tungependa uje nyumbani kwetu."
      - "Ninataka uje."
      - "Nilimwambia aende."
      - "Tutawaomba wakae."
    PurposeClausesIli:
      applies: {u-Mood: u-sbjv}
      note: "Used after 'ili' and generally in clauses with the meaning 'so that/in order that'."
      examples:
      - "Tunaondoka leo (ili) tufike huko karibuni."
      - "Mtoto alipewa kitabu asome."
    NecessityLazima:
      applies: {u-Mood: u-sbjv}
      note: "Used after (ni) lazima / lazima 'it is necessary'."
      examples:
      - "(Ni) lazima ulale."
      - "Si lazima ulale."
      - "(Ni) lazima usilale."

  Negation:
    pattern: "[SUBJ]si([OBJ])_VERB_"
    note: "Negative subjunctive is formed by inserting -si- into the tense slot; the positive subject prefix is used rather than the negative."
    examples:
    - "Tunaondoka leo (ili) tusichelewe kufika."
    - "Asiende."
    - "Nisile."
    - "Usiandike."
    - "Tusisahau."
    - "Nilimwambia asiende."
    - "Tutawaomba wasikae."
    - "(Ni) lazima usilale."
    - "Usitoke."

  Rules:
    u-Mood_u-sbjv__u-Polarity_u-pos:
      applies: {u-Mood: u-sbjv, u-Polarity: u-pos}
      patterns:
      - pattern: "[SUBJ]([OBJ])_STEM_e"
        note: "Subjunctive has no Tense-Aspect-Mood prefix and changes final -a to -e."
        examples:
        - "Aende."
        - "Nile."
        - "Uandike."
        - "Utoke."
      - pattern: "[SUBJ]([OBJ])_STEM"
        note: "Loan verbs that do not end in -a do not change; form the subjunctive by omitting any tense marker."
        examples:
        - "[SUBJ]___"
        - "[SUBJ][OBJ]___"

    u-Mood_u-sbjv__u-Polarity_u-neg:
      applies: {u-Mood: u-sbjv, u-Polarity: u-neg}
      patterns:
      - pattern: "[SUBJ]si([OBJ])_STEM_e"
        note: "Insert -si- into the tense slot; keep the positive subject prefix."
        examples:
        - "Asiende."
        - "Nisile."
        - "Usiandike."
        - "Tusisahau."
        - "Usitoke."
      - pattern: "[SUBJ]si([OBJ])_STEM"
        note: "Loan verbs: no final -a > -e change; negative formed by adding -si- and omitting any tense marker."
        examples:
        - "[SUBJ]si___"
        - "[SUBJ]si[OBJ]___"

  Features:
    u-Mood: [u-sbjv]
    u-Tense: [u-prs, u-pst, u-fut]

  Usage:
    Prs:
      applies: {u-Mood: u-sbjv, u-Tense: u-prs, x-Clausetype: x-subordinate}
      note: "Used mainly in subordinate clauses (often introduced by 'que') to express non-assertive notions (wishes, orders, possibilities, etc.)."
      examples:
      - "quero que cante"
      - "supondo que cante"
      - "ele será pago, cante ou não"
    Pst:
      applies: {u-Mood: u-sbjv, u-Tense: u-pst, x-Clausetype: x-subordinate}
      note: "Used in adverbial subordinate clauses introduced by 'se' (or equivalent) expressing conditions for a main clause in the conditional tense; also used in noun clauses introduced by 'que' as the object of past wishes or commands."
      examples:
      - "se cantasse, seria famoso"
      - "se cantasse, teríamos aplaudido"
      - "esperávamos que cantasse"
      - "eu mandei que cantassem"
    Fut:
      applies: {u-Mood: u-sbjv, u-Tense: u-fut, x-Clausetype: x-subordinate}
      note: "Used in adverbial subordinate clauses usually introduced by 'se' or 'quando', or in adjectival subordinate clauses expressing a neutral or expected condition for a present- or future-tense main clause."
      examples:
      - "se cantarmos, seremos pagos"
      - "se cantarmos, ele fica"
      - "quando cantarmos, ele escutará"

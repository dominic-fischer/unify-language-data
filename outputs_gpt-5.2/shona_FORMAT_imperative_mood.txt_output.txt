Imperative:
  Features:
    u-Mood: [u-imp]
    u-Number: [u-sg, u-pl]
    u-Politeness: [u-low, u-high]

  Rules:
    u-Mood_u-imp__u-Number_u-sg__u-Politeness_u-low:
      applies: {u-Mood: u-imp, u-Number: u-sg, u-Politeness: u-low}
      pattern: "_verb-stem_"
      note: "Used for commands to a lower-ranked person in singular."

    u-Mood_u-imp__u-Number_u-pl:
      applies: {u-Mood: u-imp, u-Number: u-pl}
      pattern: "_verb-stem_ + i"
      note: "Used for commands to multiple people."

    u-Mood_u-imp__u-Politeness_u-high:
      applies: {u-Mood: u-imp, u-Politeness: u-high}
      pattern: "_verb-stem_ + i"
      note: "Used for commands to a higher-ranked or respected person."

    u-Mood_u-imp__x-Process_x-prefixation:
      applies: {u-Mood: u-imp, x-Process: x-prefixation}
      pattern: "i- + _verb-stem_"
      note: "With monosyllabic verbs, an i- prefix is attached before the verb stem."

  Notes:
  - "The imperative has no dedicated negation; it uses the negation of the polite request form."

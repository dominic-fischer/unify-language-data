  Rules:
    u-Gender_u-bantu1-23__u-Number_u-sg__u-Part-of-Speech_u-n__x-Process_x-prefixation:
      applies:
        u-Gender: u-bantu1-23

      - pattern: ka- + _stem_

    u-Gender_u-bantu1-23__u-Number_u-pl__u-Part-of-Speech_u-n__x-Process_x-prefixation:
      applies:
        u-Gender: u-bantu1-23
        u-Number: u-pl
        u-Part-of-Speech: u-n
        x-Process: x-prefixation
      patterns:
      - pattern: tu- + _stem_
        note: diminutive class prefix (plural)
      - pattern: ti- + _stem_
        note: diminutive class prefix (plural)
    u-Part-of-Speech_u-v__u-Voice_u-pass__x-Process_x-suffixation:
      applies:
        u-Part-of-Speech: u-v
        u-Voice: u-pass
        x-Process: x-suffixation
      pattern: _stem_ + passive suffix (must be last)
      examples:
      - -andika → -andikwa
      - -andikisha → -andikishwa
      - -andikia → -andikiwa
      - -andikianisha → -andikianishwa
      - -likianisha → -likianishwa
      note: passive suffix must always be last in Swahili when suffixes are stacked
  Notes:
  - In diminutive/augmentative formation, the added syllables may attach either to
    the stem or to an already existing prefix; in some languages this can lead to
    'double' plural marking (both diminutive/augmentative and regular class prefixes
    pluralized).

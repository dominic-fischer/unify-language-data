Negation:
  Features:
    x-Negation-type: [x-clause, x-constituent]

  Rules:
    x-Negation-type_x-clause:
      applies: {x-Negation-type: x-clause}
      patterns:
        - pattern: "A- + subject marker; with 3SG (class 1/2): -ka- + verb stem"
          examples:
            - "Uhambahamba."
            - "Akahambihamba."
          note: "General negation prefix is A-. In negation, the subject marker of 3SG (class 1/2) is -ka-."
        - pattern: "subject marker + -nga- + (object marker) + verb stem (final -i)"
          examples:
            - "(Ngicela) ninganginiki isinkwa."
          note: "Used for wishes/polite requests derived from a negated verb; given as: Kennsilbe – nga – (Objektsilbe) – Verbstamm auf -i."
        - pattern: "Mu – sa + infinitive"
          note: "Negation pattern with infinitive."
        - pattern: "Mu – sa – ni + infinitive"
          note: "Negation pattern with infinitive."

    x-Negation-type_x-constituent:
      applies: {x-Negation-type: x-constituent}
      patterns:
        - pattern: "subject marker + -nge- + na/ne/no + noun"
          examples:
            - "Ungumuntu ungenomusa."
          note: "Nominal adjective negation pattern."
        - pattern: "a- + subject marker + predicative adjective"
          note: "Predicative adjective constructions can be negated with this formula; exception: in class 7/8 the subject marker changes to -ka-."
        - pattern: "Ngaphandle kwa/kwe/kwo + noun"
          examples:
            - "Ugijima ngaphandle kwejubane."
          note: "Nominal adverb negation pattern."

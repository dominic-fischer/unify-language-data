Passive:
  Features:
    u-Voice: [u-act, u-pass]
    x-Passive-type: [x-process]
    x-Clausetype: [x-main]

  Rules:
    u-Voice_u-pass__x-Passive-type_x-process:
      applies: {u-Voice: u-pass, x-Passive-type: x-process}
      patterns:
      - pattern: "ser (finite; same mood and tense as the active verb) + {past-participle of lexical verb} (+ por + {agent})"
        examples:
        - "O queijo foi comido pelo rato."
        - "A ária será cantada por aquela senhora."
        - "Se a ária fosse cantada por você, ele ficaria."
        note: "Periphrastic passive: the original direct object becomes the subject; the original subject may appear as an adverbial complement introduced by por."
      - pattern: "se + {finite verb} (3rd person; agent not expressed)"
        examples:
        - "Fizeram-se planos e criaram-se esperanças."
        note: "\"Reflexive\" passive used for third-person objects when the agent is not expressed."
      - pattern: "se + {finite verb} (intransitive; subject not expressed)"
        examples:
        - "Comeu-se, bebeu-se e bailou-se."
        note: "Extends to some intransitive verbs as an impersonal construction (subject not expressed)."

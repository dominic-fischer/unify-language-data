    PluralSpellingChanges:
      applies: {}
      patterns:
      - title: "-io (/jo/) nouns"
        pattern: "most masculine words ending in -io pronounced /jo/ drop -o and thus end in -i in the plural"
        examples:
        - "vecchio → vecchi"
        - "funzionario → funzionari"
        - "esempio → esempi"
      - title: "-cio/-gio nouns"
        pattern: "words in -cio and -gio form plurals in -ci and -gi"
        examples:
        - "bacio → baci"
      - title: "-cia/-gia nouns (rule of thumb)"
        pattern: "words in -cia and -gia form plurals in -ce/-ge if the letter before the suffix is a consonant, and in -cie/-gie if the letter before the suffix is a vowel"
        examples:
        - "frangia → frange"
        - "faccia → facce"
        - "camicia → camicie"
        - "ciliegia → ciliegie"
        notes:
        - "when the i is stressed, it always remains in the plural."
      - title: "-co/-go nouns (rule of thumb)"
        pattern: "plurals are often -chi/-ghi if the letter before the suffix is a consonant or a stressed vowel, and often -ci/-gi if it is an unstressed vowel"
        examples:
        - "fungo → funghi"
        - "stecco → stecchi"
        - "mago → maghi"
        - "fuoco → fuochi"
        - "comico → comici"
        - "medico → medici"
        notes:
        - "there are exceptions."
      - title: "-logo nouns (rule of thumb)"
        pattern: "in words ending with -logo, the plural is usually -gi when it means 'expert/student' and -ghi when it means 'speech/reasoning'"
        examples:
        - "archeologo → archeologi"
        - "catalogo → cataloghi"
      - title: "-ca/-ga nouns"
        pattern: "words in -ca and -ga form plurals in -che and -ghe"
        examples:
        - "amica → amiche"

    u-Case_u-loc__u-Gender_u-bantu1-23:
      applies: {u-Gender: u-bantu1-23, u-Case: u-loc}
      patterns:
      - title: "Class 1/2 (incl. 1a/2a) locative"
        pattern: "For nouns in class 1(a) or 2(a), form the locative by prefixing kú- and dropping the augment."
        examples:
        - "umúntu → kumúntu"
        - "abántu → kubántu"
        - "úbabá → kúbabá"
        - "ṓbabá → kṓbabá"
      - title: "Other classes locative"
        pattern: "For nouns not in class 1 or 2, form the locative by replacing the augment with e- (or o- for class 11); most nouns also take the suffix -ini, with stem-final vowel changes."
        examples:
        - "uḿlenze → emlénzeni"
        - "ámánzi → émánzini"
        - "ū́phahla → ōpháhleni"
        notes:
        - "Some nouns form a locative without the suffix, using only the locative prefix."
        - "The locative can translate as 'in/at/on/to/from' depending on context."
      endingparadigms:
      - title: "Locative suffix allomorphy with -ini"
        endingparadigm:

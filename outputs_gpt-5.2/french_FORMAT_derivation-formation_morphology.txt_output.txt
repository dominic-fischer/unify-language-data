FrenchInterrogativeFormation:
  Features:
    x-Clausetype:
    - x-int-yes-no
    - x-int-wh
    x-Word-order:
    - x-V1

  Rules:
    x-Clausetype_x-int-yes-no:
      applies: {x-Clausetype: x-int-yes-no}
      patterns:
      - pattern: "Declarative sentence + rising intonation"
        examples:
        - "Elle va rester ici ?"
        - "Je peux mettre mes photos au mur ?"
        note: "simplest and most informal; common in informal spoken French; relatively uncommon in formal spoken French or written French"
      - pattern: "Est-ce que + declarative sentence"
        examples:
        - "Est-ce qu'elle va rester ici ?"
        - "Est-ce que je peux mettre mes photos au mur ?"
        note: "can be used in any style (formal/informal, spoken/written)"
      - pattern: "..., n'est-ce pas ?"
        examples:
        - "Tu es Jane, n'est-ce pas ?"
        - "Tu parles anglais, n'est-ce pas ?"
        note: "tag-question strategy; pronounced [n‿ɛs pɑ]"
      - pattern: "Subject–verb inversion"
        examples:
        - "Es-tu content ?"
        - "Pierre est-il content ?"
        note: "typically formal and written; if subject is an unstressed pronoun, it swaps with the verb; otherwise add an unstressed subject pronoun after the verb agreeing with the subject"
    x-Clausetype_x-int-yes-no__x-Word-order_x-V1:
      applies: {x-Clausetype: x-int-yes-no, x-Word-order: x-V1}
      patterns:
      - pattern: "Verb-Subject inversion (unstressed pronoun subject)"
        examples:
        - "Es-tu content ?"
      - pattern: "Verb + agreeing unstressed subject pronoun (non-pronoun subject)"
        examples:
        - "Pierre est-il content ?"
      - pattern: "T-insertion in inversion when two vowels would be adjacent"
        examples:
        - "A-t-il 17 ans ?"
        note: "inserts ⟨t⟩ between adjacent vowels created by inversion"
      - pattern: "Inversion with 'je' avoided outside the most formal French; use est-ce que instead"
        examples:
        - "*Mens-je ?"
        - "*Prends-je le bus ?"
        note: "examples given as ungrammatical"
    x-Clausetype_x-int-wh:
      applies: {x-Clausetype: x-int-wh}
      patterns:
      - pattern: "Replace an item with a question word/phrase + rising intonation (question word may be sentence-initial or sentence-final)"
        examples:
        - "L'étudiant(e) téléphonera à son député quand ?"
        - "Qui téléphonera à son député demain ?"
        note: "simplest and generally most informal; question word/phrase position depends on what is replaced"
      - pattern: "Replace an item with a question word/phrase + front the question word/phrase"
        examples:
        - "Qui vous avez vu ?"
        - "À qui Marcel a écrit ?"
        note: "common and informal"
      - pattern: "Fronted question word/phrase + est-ce que + declarative order"
        examples:
        - "Qui est-ce que vous avez vu ?"
        - "À qui est-ce que Marcel a écrit ?"
        note: "may be used in all styles of French"
      - pattern: "Fronted question word/phrase + subject–verb inversion"
        examples:
        - "Qui avez-vous vu ?"
        - "Qui Robert a-t-il rencontré ?"
        note: "typically the most formal; in written and formal spoken French; pronoun subjects invert with the verb, otherwise add an unstressed agreeing subject pronoun after the verb"

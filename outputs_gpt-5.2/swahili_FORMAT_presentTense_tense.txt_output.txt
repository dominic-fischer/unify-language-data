    sw-Verb-type: ["sw-common-verb", "sw-short-verb", "sw-loan-verb"]

  Usage:
    sw-definite-present:
      applies: {u-Tense: u-prs, sw-Tense-type: "sw-definite-present"}
      note: "Sometimes termed the 'definite present' and formed with -na-."
      examples:
      - "Anakwenda."
      - "Unaandika."
      - "Tunajibu."
      - "(Ni)nakula."
    sw-indefinite-present:
      applies: {u-Tense: u-prs, sw-Tense-type: "sw-indefinite-present"}
      note: "Sometimes termed the 'indefinite present' and formed with -a-. In modern standard Swahili there is no great difference in meaning from the -na- present; this form is more or less obsolete and rarely used, except frequently in media headlines."
      examples:
      - "Aenda."
      - "Waandika."
      - "Twajibu."
      - "Nala."
    sw-present-habitual:
      applies: {u-Tense: u-prs, sw-Tense-type: "sw-present-habitual"}
      note: "Habitual present formed with hu-."
      examples:
      - "hu-enda"
      - "Huandiki."

  Negation:
    pattern: "[NEG.SUBJ]([OBJ]) + _stem_ (+ -i for verbs with final -a)"
    note: "No distinction is made in the negative between the -na- ('definite') and -a- ('indefinite') present: both are negated the same way."
    examples:
    - "Si endi."
    - "Haendi."
    - "Sili."
    - "Hatujibu."

Future:
  Features:
    it-Tense-type: [it-futuro-semplice, it-futuro-anteriore]

  Usage:
    it-futuro-semplice:
      applies: {u-Tense: u-fut}
      note: "Simple future (futuro semplice)."
      examples:
        - "farò — 'I will do'"

    it-futuro-anteriore:
      applies: {u-Tense: u-fut, u-Aspect: u-prf}
      note: "Future perfect (futuro anteriore); expressed as a compound tense."
      examples:
        - "avrò fatto — 'I will have done / I may have done'"

  Rules:
    it-Tense-type_it-futuro-semplice:
      applies: {it-Tense-type: it-futuro-semplice}
      forms:
        - features: {u-Mood: u-ind, u-Tense: u-fut}
          form: "farò"
      examples:
        - "farò"

    it-Tense-type_it-futuro-anteriore:
      applies: {it-Tense-type: it-futuro-anteriore}
      pattern: "auxiliary (avere/essere, future) + past participle"
      examples:
        - "avrò fatto"
      note: "Compound tenses expressing perfect aspect are formed with either auxiliary avere or essere plus the past participle; the auxiliary choice is important for past participle agreement."

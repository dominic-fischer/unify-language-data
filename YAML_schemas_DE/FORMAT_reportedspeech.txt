ReportedSpeech:
  Features:
    Mode: [direct, indirect]
    ClauseType: [declarative, interrogative, exclamative]
    ReportedTense: [pres, past, fut]
    Mood: [indicative, subjI, subjII]

  Rules:
    DirectSpeech:
      applies: {mode: direct}
      pattern: "{quote} + {reporting verb}"
      example: "Er sagte: 'Ich komme morgen.'"
      note: "quotation preserves original tense/mood exactly"

    IndirectDeclarative:
      applies: {mode: indirect, clauseType: declarative}
      pattern:
        - {reportedTense: pres, pattern: "{reporting verb} + dass/Ø + {clause in Subjunctive I (present)}", example: "Er sagte, er komme morgen.", note: "reported present"}
        - {reportedTense: past, pattern: "{reporting verb} + dass/Ø + {clause in Subjunctive I (perfect)}", example: "Er sagte, er sei gestern gekommen.", note: "reported past"}
        - {reportedTense: fut, pattern: "{reporting verb} + dass/Ø + {clause in Subjunctive I (future)}", example: "Er sagte, er werde morgen kommen.", note: "reported future"}

    IndirectInterrogative:
      applies: {mode: indirect, clauseType: interrogative}
      pattern: "{reporting verb} + ob/wh-word + {clause in Subjunctive I}"
      example: "Er fragte, ob er morgen komme."
      note: "yes/no questions use 'ob', wh-questions use wh-word; tense follows same system as declaratives (pres, past, fut)"

    IndirectExclamative:
      applies: {mode: indirect, clauseType: exclamative}
      pattern: "{reporting verb} + wie/was für + {clause in Subjunctive I}"
      example: "Er sagte, wie schön es sei!"
      note: "exclamatives follow the same tense distinctions as declaratives; Subjunctive I preferred, Subjunctive II if forms coincide"

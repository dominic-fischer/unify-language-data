  Features:
    mood: [imp]
    person: [first, second, third]
    number: [sg, pl]

  Negation:
    pattern: "[Vfin (...)] nicht (...)"
    example: "Geh doch nicht dahin!"

  Rules:
    mood-imp_number-sg_person-second:
      applies: {mood: imp, person: second, number: sg}
      pattern: "{stem} + (optionally -e)"
      example: "geh!, mache!"
      note: "direct imperative singular; -e ending optional in 2sg, often dropped in colloquial usage"

    mood-imp_number-pl_person-second:
      applies: {mood: imp, person: second, number: pl}
      pattern: "{stem} + t"
      example: "geht!, macht!"
      note: "direct imperative plural"

      applies: {mood: imp, person: first, number: pl}

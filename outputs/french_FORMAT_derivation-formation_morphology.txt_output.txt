FrenchQuestionFormation:
  Features:
    u-Interrogativity:
    - u-int
  Rules:
    u-Interrogativity_u-int:
      applies: {u-Interrogativity: u-int}
      patterns:
        - pattern: "Intonation-only yes/no question on declarative sentence"
          examples:
            - "Elle va rester ici ?"
            - "Je peux mettre mes photos au mur ?"
          note: "Yes/no questions formed by rising intonation; informal spoken French."
        - pattern: "Est-ce que + declarative sentence"
          examples:
            - "Est-ce qu'elle va rester ici ?"
            - "Est-ce que je peux mettre mes photos au mur ?"
          note: "Can be used in any style; formal, informal, spoken, or written."
        - pattern: "Inversion of subject and verb"
          examples:
            - "Es-tu content ?"
            - "Pierre est-il content ?"
            - "A-t-il 17 ans ?"
            - "Mens-je ?"
            - "Prends-je le bus ?"
          note: "Inversion used in formal and written French; t-insertion occurs between vowels; je-inversion restricted to the most formal French; est-ce-que increasingly common."
        - pattern: "Wh-questions and information questions"
          examples:
            - "L'étudiant(e) téléphonera à son député demain."
            - "L'étudiant(e) téléphonera à son député quand ?"
            - "Qui téléphonera à son député demain ?"
            - "Qui vous avez vu ?"
            - "À qui Marcel a écrit ?"
            - "Qui est-ce que vous avez vu ?"
            - "À qui est-ce que Marcel a écrit ?"
            - "Qui avez-vous vu ?"
            - "Qui Robert a-t-il rencontré ?"
          note: "Information questions formed by question word substitution/fronting and est-ce que insertion; can also involve subject–verb inversion in the most formal cases."

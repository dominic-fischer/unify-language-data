  Features:
    u-Aspect: [u-prf, u-prosp]
    u-Polarity: [u-neg, u-pos]
    u-Tense: [u-fut]

  Rules:
    u-Polarity_u-pos__u-Tense_u-fut:
      applies: {u-Tense: u-fut, u-Polarity: u-pos}
      patterns:
      - pattern: "SUBJ – ta – STEM"
        examples:
        - "wa – ta – kula"
        - "Atakwenda."
        - "Nitakula."
        - "Utaandika."
        - "Tutajibu."

    u-Polarity_u-neg__u-Tense_u-fut:
      applies: {u-Tense: u-fut, u-Polarity: u-neg}
      patterns:
      - title: "usual"
        pattern: "NEG.SUBJ – ta – STEM"
        examples:
        - "hawa – ta – kula"
        - "Hatakwenda."
        - "Sitakula."
        - "Hutaandika."
        - "Hatutajibu."
      - title: "variant"
        pattern: "NEG.SUBJ – to – STEM"
        notes:
        - "In the negative future, some speakers use -to- instead of -ta-."
        examples:
        - "hatokuja"

    "Future perfect":
      title: "Future perfect"
      applies: {u-Aspect: u-prf, u-Tense: u-fut, u-Polarity: u-neg}

        pattern: "SUBJ – ta – kuwa NEGSUBJ – ja – STEM"

      - title: "auxiliary negated"
        pattern: "NEGSUBJ – ta – kuwa SUBJ – me – STEM"
        examples:

      patterns:
      - pattern: "Verb: SubjectMarker-(TenseMarker)-(AspectMarker)-VerbStem"
        notes:

        examples:

        notes:
        - "Compound tenses occur with an auxiliary-like verb form plus an additional finite verb form."
        examples:
        - "nd-a-khala ndí-kú-gúla 'I have been buying'"

Adverbs:
  Features:
    u-Comparison: [u-cmpr, u-sprl, u-rl, u-ab]
    u-Part-of-Speech: [u-adv, u-adj]

  Rules:
    u-Part-of-Speech_u-adv:
      applies: {u-Part-of-Speech: u-adv}
      pattern: "lexical base form"
      examples: 
      - "gestern"
      - "oft"
      - "bald"

    u-Part-of-Speech_u-adj:
      applies: {u-Part-of-Speech: u-adj}
      pattern: "adjective stem reused as adverb (uninflected)"
      examples: 
      - "schnell → schnell"
      - "laut → laut"
      note: "form identical to adjective"

    u-Comparison_u-cmpr:
      applies: {u-Comparison: u-cmpr}
      pattern: "_stem_ + er"
      examples: 
      - "schnell → schneller"

    u-Comparison_u-sprl_u-ab:
      applies: {u-Comparison: [u-sprl, u-ab]}
      pattern: "am _stem_ + sten"
      examples: 
      - "schnell → am schnellsten"

    u-Comparison_u-sprl_u-rl:
      applies: {u-Comparison: [u-sprl, u-rl]}
      pattern: "Determiner _stem_ + ste"
      examples:
      - "schnell → schnellste"

ReportedSpeech:
  Features:
    u-Evidentiality: [u-fh, u-quot]
    u-Tense: [u-prs, u-pst, u-fut]

  Notes:
  - "The tense of the subordinate clause remains unchanged unless the reporting verb is in the past, in which case the reported tense is adjusted relative to it as outlined."
  - "In either case, though, mood changes: Subjunctive I is the mood of reporting, Subjunctive II if forms coincide"
  - "Using 'dass' will trigger the inversion, as the sentence then becomes a subclause."

  Rules:
    u-Evidentiality_u-fh:
      applies: {u-Evidentiality: u-fh}
      pattern: "quote + _reporting verb_"
      example: "Er sagte: 'Ich komme morgen.'"
      note: "quotation preserves original tense/mood exactly"

    u-Evidentiality_u-quot__u-Tense_u-prs:
      applies: {u-Evidentiality: u-quot, u-Interrogativity: u-decl, u-Tense: u-prs}
      patterns: 
      - pattern: "_reporting verb_ + dass/Ø + {clause in Subjunctive I (present)}"
        examples:
        - "'Ich komme morgen' --> Er sagte, er komme morgen."
        - "'Ich komme morgen' --> Er sagte, dass er morgen komme."
      - pattern: "_reporting verb_ + ob/wh-word + {clause in Subjunctive I (present)}"
        examples: 
        - "'Kommst du morgen?' --> Er fragte, ob ich morgen komme/käme."
        - "'Wann kommst du morgen?' --> Er fragte, wann ich morgen komme/käme."
        note: "yes/no questions use 'ob', wh-questions use wh-word; tense follows same system as u-decls (pres, past, fut)"
      - pattern: "_reporting verb_ + wie/was/dass für + {clause in Subjunctive I (present)}"
        examples: 
        - "'Ich freue mich so sehr!' --> Er brachte zum Ausdruck, wie sehr er sich freue."
        - "'Ich freue mich so sehr!' --> Er sagte, dass er sich sehr freue."
        notes:
        - "'Sagen' does not license reported speech with 'wie'.

    u-Evidentiality_u-quot__u-Tense_u-pst:
      applies: {u-Evidentiality: u-quot, u-Tense: u-pst}
      pattern: "_reporting verb_ + dass/Ø + {clause in Subjunctive I (perfect)}"
      examples: 
      - "Er sagte, er sei gestern angekommen."
      - "Ich fragte, ob er gestern angekommen sei."
      - "Er erzählte, wie schön es gewesen sei."

    u-Evidentiality_u-quot__u-Tense_u-fut:
      applies: {u-Evidentiality: u-quot, u-Tense: u-fut}
      pattern: "_reporting verb_ + dass/Ø + {clause in Subjunctive I (future)}"
      examples: 
      - "Er sagte, er werde morgen kommen."
      - "Ich fragte, ob er nächste Woche ankommen werde."
      - "'Es wird so schön (werden)!" --> Er meinte, dass es sehr schön werde/werden würde."
